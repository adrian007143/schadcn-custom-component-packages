# Form Layout

Composable layout primitives for structuring forms. Use these when you want full control over form structure without using `FormBuilderStandard`.

## Full example

```tsx
import { FormLayout } from "@/components/forms/form-layout/FormLayout"
import { FormSection } from "@/components/forms/form-layout/FormSection"
import { FormColumns } from "@/components/forms/form-layout/FormColumns"
import { FormDivider } from "@/components/forms/form-layout/FormDivider"
import { FormActions } from "@/components/forms/form-layout/FormActions"

export function AddressForm() {
  return (
    <FormLayout>
      <FormSection title="Account" description="Your login details.">
        <FormColumns columns={2}>
          <DynamicFormField control={control} name="email" fieldType={FormFieldType.INPUT} label="Email" />
          <DynamicFormField control={control} name="phone" fieldType={FormFieldType.PHONE_INPUT} label="Phone" />
        </FormColumns>
      </FormSection>

      <FormDivider />

      <FormSection title="Address">
        <FormColumns columns={3}>
          <DynamicFormField control={control} name="city" fieldType={FormFieldType.INPUT} label="City" />
          <DynamicFormField control={control} name="state" fieldType={FormFieldType.INPUT} label="State" />
          <DynamicFormField control={control} name="zip" fieldType={FormFieldType.MASKED} label="ZIP" mask="99999" />
        </FormColumns>
      </FormSection>

      <FormActions layout="right">
        <Button variant="outline">Cancel</Button>
        <Button type="submit">Save</Button>
      </FormActions>
    </FormLayout>
  )
}
```

## FormLayout

Top-level wrapper. Applies `space-y-10` between child sections.

```tsx
<FormLayout className="max-w-2xl">
  {/* sections */}
</FormLayout>
```

## FormSection

Groups related fields under an optional title and description.

  { name: "title", type: "string", description: "Section heading rendered above the fields" },

## FormColumns

Renders a responsive CSS grid. On screens smaller than `sm`, always collapses to a single column.

<PropsTable items={[
  { name: "columns", type: "1 | 2 | 3 | 4", default: "2", description: "Number of equal-width columns" },
  { name: "className", type: "string", description: "Extra class applied to the grid container" },
  { name: "children", type: "ReactNode", required: true, description: "Fields placed in grid cells" },
]} />

<Callout variant="tip">
  On small screens, `FormColumns` always collapses to a single column regardless of the `columns` prop — no media query overrides needed.
</Callout>

## FormRow

Horizontal flex row for fields that sit side-by-side at equal widths.

```tsx
<FormRow>
  <DynamicFormField ... name="startDate" label="Start" />
  <DynamicFormField ... name="endDate" label="End" />
</FormRow>
```

## FormActions

Renders a top separator followed by a flex row for action buttons.

<PropsTable items={[
  { name: "layout", type: '"stack" | "inline" | "right" | "space-between"', default: '"right"', description: "Button alignment within the action bar" },
  { name: "children", type: "ReactNode", required: true, description: "Button components" },
]} />

```tsx
<FormActions layout="space-between">
  <Button variant="outline">Back</Button>
  <Button type="submit">Next</Button>
</FormActions>
```

## FormDivider

Visual `<hr>` separator between sections.

```tsx
<FormDivider />
```
