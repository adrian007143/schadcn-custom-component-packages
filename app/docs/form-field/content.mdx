# DynamicFormField

A single component that renders any of 17+ field types based on the `fieldType` prop. Built on `react-hook-form` — just pass `control` and `name` from your form.

## Install

<InstallCmd file="form-field.json" />

## Usage

```tsx
import { useForm } from "react-hook-form"
import { Form } from "@/components/ui/form"
import { DynamicFormField } from "@/components/forms/form-field/DynamicFormField"
import { FormFieldType } from "@/components/forms/form-field/constants"

export function MyForm() {
  const form = useForm({ defaultValues: { name: "" } })

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(console.log)}>
        <DynamicFormField
          control={form.control}
          name="name"
          fieldType={FormFieldType.INPUT}
          label="Full name"
          placeholder="John Doe"
          required
        />
      </form>
    </Form>
  )
}
```

<Callout variant="tip">
  Use **FormBuilderStandard** if you want to define an entire form from a config array — it uses `DynamicFormField` internally. See the [FormBuilderStandard docs](/docs/form-builder).
</Callout>

## Examples

### Async select

```tsx
<DynamicFormField
  control={form.control}
  name="userId"
  fieldType={FormFieldType.ASYNC_SELECT}
  label="Assign to"
  data={[]}
  labelKey="name"
  valueKey="id"
  loadOptions={async (query) => {
    const res = await fetch(`/api/users?q=${query}`)
    return res.json()
  }}
/>
```

### Currency input

```tsx
<DynamicFormField
  control={form.control}
  name="price"
  fieldType={FormFieldType.CURRENCY}
  label="Price"
  currencyLocale="en-US"
  decimalPlaces={2}
  thousandSeparator
/>
```

### Masked input

```tsx
<DynamicFormField
  control={form.control}
  name="phone"
  fieldType={FormFieldType.MASKED}
  label="Phone"
  mask="(999) 999-9999"
/>
```

### With prefix / suffix

```tsx
<DynamicFormField
  control={form.control}
  name="website"
  fieldType={FormFieldType.INPUT}
  label="Website"
  prefix={<span className="text-muted-foreground text-sm">https://</span>}
/>
```
