# FieldRenderer

A single component that renders any of 25+ field types based on the `fieldType` prop. Built on `react-hook-form` — just pass `control` and `name` from your form.

## Install

<InstallCmd file="form-field.json" />

<Callout variant="info">
  This single command installs all npm packages (`react-hook-form`, `@hookform/resolvers`, `zod`, `react-phone-number-input`) and all required shadcn components (`field`, `button`, `sonner`, and all field primitives) automatically.
</Callout>

## Usage

<Callout variant="warning">
  Form components use React hooks — add `"use client"` to any file that renders a form.
</Callout>

```tsx
"use client"

import { useForm } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import { z } from "zod"
import { Button } from "@/components/ui/button"
import { FieldRenderer } from "@/components/forms/core"
import { FormFieldType } from "@/components/forms/core"

const schema = z.object({
  email: z.string().email(),
  password: z.string().min(8),
})

export function LoginForm() {
  const form = useForm({
    resolver: zodResolver(schema),
    defaultValues: { email: "", password: "" },
  })

  return (
    <form onSubmit={form.handleSubmit(console.log)} className="space-y-4">
      <FieldRenderer
        control={form.control}
        name="email"
        fieldType={FormFieldType.INPUT}
        label="Email"
        placeholder="you@example.com"
        required
      />
      <FieldRenderer
        control={form.control}
        name="password"
        fieldType={FormFieldType.PASSWORD}
        label="Password"
        required
      />
      <Button type="submit">Sign in</Button>
    </form>
  )
}
```

<Callout variant="tip">
  Want to skip the boilerplate? Use **SchemaForm** — it handles schema, layout, and submission automatically. See the [SchemaForm docs](/docs/form-builder).
</Callout>

## Examples

### Async select

```tsx
<FieldRenderer
  control={form.control}
  name="userId"
  fieldType={FormFieldType.ASYNC_SELECT}
  label="Assign to"
  data={[]}
  labelKey="name"
  valueKey="id"
  loadOptions={async (query) => {
    const res = await fetch(`/api/users?q=${query}`)
    return res.json()
  }}
/>
```

### Currency input

```tsx
<FieldRenderer
  control={form.control}
  name="price"
  fieldType={FormFieldType.CURRENCY}
  label="Price"
  currencyLocale="en-US"
  decimalPlaces={2}
  thousandSeparator
/>
```

### Masked input

```tsx
<FieldRenderer
  control={form.control}
  name="phone"
  fieldType={FormFieldType.MASKED}
  label="Phone"
  mask="(999) 999-9999"
/>
```

### Slider

```tsx
<FieldRenderer
  control={form.control}
  name="rating"
  fieldType={FormFieldType.SLIDER}
  label="Rating"
  sliderMin={0}
  sliderMax={10}
  sliderStep={1}
  showSliderValue
/>
```

### Conditional field

```tsx
<FieldRenderer
  control={form.control}
  name="companyName"
  fieldType={FormFieldType.INPUT}
  label="Company name"
  show={(values) => values.accountType === "business"}
/>
```

### With prefix / suffix

```tsx
<FieldRenderer
  control={form.control}
  name="website"
  fieldType={FormFieldType.INPUT}
  label="Website"
  prefix={<span className="text-muted-foreground text-sm">https://</span>}
/>
```
