# Redux Setup

A complete Redux configuration with `react-redux-methods`, localStorage persistence, and ready-to-use example slices. One install gives you a working store with DevTools support.

## What's installed

```
redux/
├── store.ts              Redux store with DevTools + localStorage persistence
├── StoreProvider.tsx     React context wrapper component
├── connections.ts        Typed useSelector / useDispatch hooks
└── reducers/
    ├── index.ts          Root reducer combining all slices
    ├── notification.ts   Show/hide notification state
    ├── sidebar.ts        Sidebar open/close state
    └── todo.ts           Todo list with add/toggle/delete

lib/helper/storageLocal.ts  localStorage read/write helpers
```

## Setup

Wrap your app root with `StoreProvider`:

```tsx
// app/layout.tsx
import StoreProvider from "@/redux/StoreProvider"

export default function RootLayout({ children }) {
  return (
    <html>
      <body>
        <StoreProvider>{children}</StoreProvider>
      </body>
    </html>
  )
}
```

<Callout variant="info">
  `StoreProvider` is a `"use client"` component. It's safe to use in your root layout — it will only hydrate on the client side.
</Callout>

## Reading state

Use the typed selector hooks:

```tsx
"use client"

import { useSelector } from "@/redux/connections"
import { getTodos, getTodoCount } from "@/redux/reducers/todo"
import { getNotificationStatus } from "@/redux/reducers/notification"
import { getSidebarStatus } from "@/redux/reducers/sidebar"

export function StatusBar() {
  const todos = useSelector(getTodos)
  const count = useSelector(getTodoCount)
  const notifOpen = useSelector(getNotificationStatus)
  const sidebarOpen = useSelector(getSidebarStatus)

  return <div>{count} todos · {notifOpen ? "Notif open" : "Notif closed"}</div>
}
```

## Dispatching actions

```tsx
"use client"

import { useDispatch } from "@/redux/connections"
import { addTodo, toggleTodo, deleteTodo } from "@/redux/reducers/todo"
import { setNotificationShow, setNotificationHide } from "@/redux/reducers/notification"
import { setSidebarShow, setSidebarHide } from "@/redux/reducers/sidebar"

export function TodoActions() {
  const dispatch = useDispatch()

  return (
    <button onClick={() =>
      dispatch(addTodo({ id: crypto.randomUUID(), text: "New task", completed: false }))
    }>
      Add Todo
    </button>
  )
}
```

## localStorage persistence

Selected slices are automatically saved to localStorage and restored on page load (SSR-safe — only runs in the browser).

Default persisted slices: `notifications`, `sidebar`, `todo`.

To add a new slice to persistence, update `redux/store.ts`:

```ts
// redux/store.ts — add to the persistence config
const persistedReducer = persistReducer({
  key: "root",
  storage,
  whitelist: ["notifications", "sidebar", "todo", "myFeature"], // ← add here
}, rootReducer)
```

## Adding a new slice

```ts
// redux/reducers/myFeature.ts
import { createSlice } from "react-redux-methods"

const myFeatureSlice = createSlice({
  name: "myFeature",
  initialState: { value: "" as string },
  reducers: {
    setValue: (state, action) => {
      state.value = action.payload
    },
    clearValue: (state) => {
      state.value = ""
    },
  },
})

export const { setValue, clearValue } = myFeatureSlice.actions
export const getMyFeatureValue = (state: RootState) => state.myFeature.value
export default myFeatureSlice.reducer
```

Then register it in `redux/reducers/index.ts`:

```ts
import myFeatureReducer from "./myFeature"

export const rootReducer = combineReducers({
  // ...existing reducers
  myFeature: myFeatureReducer,
})
```

<Callout variant="tip">
  The `createSlice` from `react-redux-methods` is compatible with standard Redux Toolkit patterns — you can mix and match with existing Redux Toolkit slices.
</Callout>
