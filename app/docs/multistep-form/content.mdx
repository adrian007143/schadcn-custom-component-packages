# MultiStepForm

A multi-step form with per-step Zod validation, progress tracking, flexible layouts, and optional reCAPTCHA. Each step validates independently — the user can't advance until the current step is valid.

## Install

<InstallCmd file="multistep-form-template.json" />

<Callout variant="warning">
  This block installs `sonner`. After running the CLI, add the toaster once in your root layout:
</Callout>

```tsx

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <body>
        {children}
        <Toaster position="top-center" />
      </body>
    </html>
  )
}
```

## Usage

```tsx
import { MultiStepForm } from "@/components/forms/form-ui/multistep-form/MultiStepForm"
import { FormFieldType } from "@/components/forms/form-field/constants"
import { z } from "zod"

const step1 = z.object({
  orgName: z.string().min(1, "Required"),
  industry: z.string().min(1, "Required"),
})

const step2 = z.object({
  adminEmail: z.string().email(),
  adminName: z.string().min(1, "Required"),
})

export function CreateOrgForm() {
  return (
    <MultiStepForm
      defaultValues={{ orgName: "", industry: "", adminEmail: "", adminName: "" }}
      steps={[
        {
          title: "Organization",
          schema: step1,
          fields: [
            { name: "orgName", fieldType: FormFieldType.INPUT, label: "Organization name" },
            { name: "industry", fieldType: FormFieldType.SELECT, label: "Industry",
              data: industries, labelKey: "label", valueKey: "value" },
          ],
        },
        {
          title: "Admin Account",
          schema: step2,
          fields: [
            { name: "adminName", fieldType: FormFieldType.INPUT, label: "Full name" },
            { name: "adminEmail", fieldType: FormFieldType.INPUT, label: "Email address" },
          ],
        },
      ]}
      onSubmit={async (values) => {
        await createOrg(values)
      }}
    />
  )
}
```

## Step config

```ts
interface MultiStepFormStep<T> {
  title: string               // Shown in the progress indicator
  description?: string        // Optional sub-title for the step
  schema: ZodType             // Validated when "Next" is clicked
  fields: FormBuilderField<T>[]
  sections?: FormBuilderSection<T>[]  // Use sections for multi-column layouts
  layout?: SectionLayout      // Grid layout for this step's fields
}
```

<Callout variant="info">
  Each step's `schema` should only include the fields present in that step. The combined values of all steps are passed to `onSubmit`.
</Callout>

## Layout modes

```tsx
// Auto — horizontal on desktop, vertical on mobile
<MultiStepForm layout="auto" ... />

// Always horizontal (steps shown left-to-right)
<MultiStepForm layout="horizontal" ... />

// Always vertical (steps shown top-to-bottom)
<MultiStepForm layout="vertical" ... />
```

## reCAPTCHA

Set `recaptcha={true}` with your site key. The reCAPTCHA appears on the final step before the submit button. `onSubmit` is only called after a valid reCAPTCHA token is obtained.

```tsx
<MultiStepForm
  recaptcha
  siteKey={process.env.NEXT_PUBLIC_RECAPTCHA_SITE_KEY}
  ...
/>
```

<Callout variant="warning">
  reCAPTCHA requires a valid Google site key. Use `NEXT_PUBLIC_RECAPTCHA_SITE_KEY` in your `.env.local` file. The component will not render the reCAPTCHA widget if no `siteKey` is provided.
</Callout>

## Per-step validation

Clicking "Next" triggers validation for only the fields in the current step's `schema`. The user cannot advance until all fields in the current step are valid. This means:

1. Each step's `schema` keys must match the field `name` values in that step
2. The combined type `T` spans all steps — use `z.object({...step1, ...step2})` for the full schema type

```ts
// Full schema type for type safety
type FormValues = z.infer<typeof step1> & z.infer<typeof step2>
```
