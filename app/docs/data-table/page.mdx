# DynamicDataTable

A fully featured, editable data table built on TanStack Table v8. Supports inline cell editing, drag-and-drop row/column reordering, faceted filtering, column visibility toggles, and custom cell renderers.

<InstallCmd file="data-table-dynamic.json" />

## Usage

```tsx
import { DynamicDataTable } from "@/components/data-table/data-table-dynamic/core/dynamic-data-table"

type User = {
  id: string
  name: string
  email: string
  role: string
  status: string
  joinedAt: string
}

const columns: DynamicColumnDef<User>[] = [
  { accessorKey: "name", header: "Name", cellType: "editable" },
  { accessorKey: "email", header: "Email", cellType: "editable" },
  { accessorKey: "role", header: "Role", cellType: "select",
    selectOptions: ["admin", "user", "viewer"] },
  { accessorKey: "status", header: "Status", cellType: "badge" },
  { accessorKey: "joinedAt", header: "Joined", cellType: "datepicker" },
]

export function UsersTable({ data }: { data: User[] }) {
  return (
    <DynamicDataTable
      data={data}
      columnConfig={columns}
      rowIdKey="id"
      enableRowSelection
      enableRowDrag
      enableColumnDrag
      enableGlobalFilter
      enableColumnVisibility
      onDataChange={(updated) => console.log("Updated rows:", updated)}
      onDeleteRow={(row) => deleteUser(row.id)}
    />
  )
}
```

## Column definition

```ts
interface DynamicColumnDef<TData> {
  accessorKey: keyof TData & string  // Maps to a data field
  header: string                     // Column heading text
  cellType?: CellType                // How to render the cell
  selectOptions?: string[]           // Options for cellType "select"
  width?: number                     // Fixed column width in px
  enableSorting?: boolean            // Allow column sorting
  enableHiding?: boolean             // Allow column to be hidden
}
```

## Drag and drop

Row and column drag-and-drop uses `@dnd-kit`. Enable with `enableRowDrag` and/or `enableColumnDrag`.

- **Row drag**: Dragging a row to a new position triggers `onDataChange` with the reordered array
- **Column drag**: Dragging a column header reorders the visible columns in-place

<Callout variant="info">
  Drag handles are shown automatically when `enableRowDrag` is true. No extra configuration needed.
</Callout>

## Faceted filtering

The table includes built-in faceted filter support for columns with a finite set of values (e.g. status, role). Filters appear as multi-select dropdowns in the toolbar.

<Callout variant="tip">
  Combine `enableGlobalFilter` (full-text search) with faceted filters for a full search experience with no extra code.
</Callout>
