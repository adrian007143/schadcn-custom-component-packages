# DataGrid

A fully featured, editable data grid built on TanStack Table v8. Supports inline cell editing, drag-and-drop row reordering, faceted column filtering, global search, column visibility, CSV export, and custom cell renderers — all in one installable block.

## Basic usage

```tsx
import { DataGrid } from "@/components/data-grid/core/DataGrid"
import type { DataGridColumn } from "@/components/data-grid/types/columns"

type User = {
  id: string
  name: string
  email: string
  role: string
  status: string
  joinedAt: Date
}

const columns: DataGridColumn<User, unknown>[] = [
  { accessorKey: "name",     header: "Full Name",  editable: true, inputType: "text" },
  { accessorKey: "email",    header: "Email",      editable: true, inputType: "text" },
  { accessorKey: "role",     header: "Role",       editable: true, inputType: "select",
    selectOptions: [{ label: "Admin", value: "admin" }, { label: "User", value: "user" }, { label: "Manager", value: "manager" }],
    filterOptions: [{ label: "Admin", value: "admin" }, { label: "User", value: "user" }, { label: "Manager", value: "manager" }],
  },
  { accessorKey: "status",   header: "Status",     inputType: "badge" },
  { accessorKey: "joinedAt", header: "Joined",     editable: true, inputType: "datepicker" },
]

export function UsersTable({ data }: { data: User[] }) {
  return (
    <DataGrid
      data={data}
      columnConfig={columns}
      rowIdKey="id"
      toolbarTitle="User Management"
      toolbarSubtitle="View and manage system access and permissions."
      enableRowSelection
      enableRowDrag
      enableGlobalFilter
      enableColumnVisibility
      onDataChange={(updated) => console.log("Updated:", updated)}
      onSave={(saved) => saveToServer(saved)}
      onDeleteRow={(row) => deleteUser(row.id)}
      onDeleteSelected={(rows) => bulkDelete(rows)}
      onAddRow={() => openAddUserDialog()}
    />
  )
}
```

<Callout variant="tip">
  The deprecated alias `DynamicDataTable` still works. New code should use `DataGrid` from `@/components/data-grid/core/DataGrid`.
</Callout>

## Column definition

```ts
import type { DataGridColumn } from "@/components/data-grid/types/columns"

const col: DataGridColumn<MyRow, unknown> = {
  accessorKey: "role",           // Maps to a field in your data type
  header: "Role",                // Column header label
  editable: true,                // Enable inline editing
  inputType: "select",           // Cell editor type (see below)
  selectOptions: [               // Required when inputType is "select"
    { label: "Admin",   value: "admin" },
    { label: "User",    value: "user" },
    { label: "Manager", value: "manager" },
  ],
  filterOptions: [               // Enables a faceted filter chip in the toolbar
    { label: "Admin",   value: "admin" },
    { label: "User",    value: "user" },
    { label: "Manager", value: "manager" },
  ],
  size: 140,                     // Fixed column width in px (optional)
  enableSorting: true,           // Allow sorting by this column
  enableHiding: true,            // Allow hiding this column
}
```

## Faceted filters

Add `filterOptions` to any column to enable a multi-select filter chip in the toolbar. Multiple values use OR logic. Combine with `enableGlobalFilter` for a full search + filter experience.

```ts
{
  accessorKey: "role",
  header: "Role",
  filterOptions: [
    { label: "Admin",   value: "admin" },
    { label: "User",    value: "user" },
    { label: "Manager", value: "manager" },
  ],
}
```

<Callout variant="info">
  The filter chip shows how many rows match each value. A **Reset** button appears in the toolbar whenever any filter or search is active.
</Callout>

## Avatar / image column

Display an image or avatar without enabling editing:

```ts
{
  accessorKey: "avatar",
  header: "Avatar",
  displayMode: "avatar",   // "avatar" | "rounded" | "square"
  imageSize: "md",         // "sm" | "md" | "lg" | "xl"
  imageAltKey: "name",     // key in your row data to use as alt text
}
```

## Drag and drop

Enable drag handles for row reordering with `enableRowDrag`. Dragging a row fires `onDataChange` with the updated array immediately — no extra wiring needed.

```tsx
<DataGrid
  data={data}
  columnConfig={columns}
  rowIdKey="id"
  enableRowDrag
  onDataChange={(rows) => setData(rows)}
/>
```

<Callout variant="info">
  Drag handles appear automatically in the first column when `enableRowDrag` is true.
</Callout>

## Inline editing + Save flow

When `onSave` is provided, the toolbar shows a **Save changes** button only when unsaved edits exist:

```tsx
const [data, setData] = React.useState(initialData)

<DataGrid
  data={data}
  columnConfig={columns}
  rowIdKey="id"
  onDataChange={setData}
  onSave={async (updated) => {
    await api.patch("/users", updated)
    toast.success("Saved")
  }}
/>
```

## Loading skeleton

Pass `isLoading` to show skeleton rows while data is being fetched:

```tsx
const { data, isLoading } = useSWR("/api/users", fetcher)

<DataGrid
  data={data ?? []}
  columnConfig={columns}
  rowIdKey="id"
  isLoading={isLoading}
/>
```

## Row actions menu

Pass any combination of `onViewRow`, `onEditRow`, `onDeleteRow` to get a `⋮` actions menu on each row. `onDeleteRow` automatically shows a confirmation dialog before firing.

```tsx
<DataGrid
  data={data}
  columnConfig={columns}
  rowIdKey="id"
  onViewRow={(row) => router.push(`/users/${row.id}`)}
  onEditRow={(row) => openEditModal(row)}
  onDeleteRow={(row) => deleteUser(row.id)}
/>
```

## CSV export

The built-in export button writes all visible, filtered rows to a `.csv` file. To handle export yourself (e.g. server-side), pass `onExport`:

```tsx
<DataGrid
  data={data}
  columnConfig={columns}
  rowIdKey="id"
  onExport={() => downloadServerCSV()}
/>
```
